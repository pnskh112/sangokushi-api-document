### 概念スキーマを元に作成したER図

```uml
@startuml
title ER図

entity 時代区分{
  * 時代区分コード
  * 国コード<<FK>>
  --
  * 時代区分名
  * 開始年度
  * 終了年度
}

entity 国{
  * 国コード
  --
  * 国名
  * 説明
}

entity 武将{
  * 武将コード
  * 国コード<<FK>>
  --
  * 武将名
  * 字
  * 人物像
  * 趣味
  * 行年・没年
}

entity 役職{
  * 役職コード
  --
  * 役職名
  * 役職説明
}

entity 就任{
  * 就任コード
  * 役職コード<<FK>>
  * 武将コード<<FK>>
  * 時代区分コード<<FK>>
  --
}

entity 画像{
  * 画像コード
  * 武将コード<<FK>>
  --
  * 商品コード
}

entity 武器兵馬{
  * 武器兵馬コード
  * 武将コード<<FK>>
  --
  * 武器兵馬名
  * 武器兵馬説明
}

entity 領地{
  * 領地コード
  * 国コード<<FK>>
  --
  * 領地名
  * 領地説明
}

entity 戦{
  * 戦コード
  * 領地コード<<FK>>
  * 時代区分コード<<FK>>
  --
  * 戦
  * 概要
}

entity 名言エピソード{
  * 名言エピソードコード
  * 戦コード<<FK>>
  * 武将コード<<FK>>
  --
  * タイトル
  * エピソード
}

国 ||..|{ 領地
国 ||..|{ 武将
国 ||..|{ 時代区分
時代区分 ||..|| 就任
武将 ||..|{ 就任
就任 ||..|| 役職
武将 }|..o{ 武器兵馬
武将 ||..|| 画像
時代区分 ||..o{ 戦
領地 ||..o{ 戦
戦 ||..o{ 名言エピソード
時代区分 ||..o{ 名言エピソード
武将 ||..o{ 名言エピソード
@enduml
```
